技术选型（先说清楚）

Node.js + ES Modules（"type": "module"）
LangChain.js：用 @langchain/openai 这个包，它支持自定义 baseURL 和模型名，可以接任何 OpenAI 兼容服务
rss-parser：拉 RSS
dotenv：加载环境变量
YAML 配置：用 yaml 这个轻量包，因为你最终是要做"多主题"的

不引入：日期库（用原生 Date）、命令行参数库（用 process.argv）、日志库（用 console）。这些等真有需求再加。

关键设计：LLM 调用封装在一个文件里
这是整个方案最重要的扩展点。src/llm.js 是唯一一处接触 LangChain 的地方，未来如果你想换模型、换 provider、加 fallback、加重试，全都改这一个文件。
src/llm.js
jsimport { ChatOpenAI } from '@langchain/openai'
import { ChatPromptTemplate } from '@langchain/core/prompts'
import { JsonOutputParser } from '@langchain/core/output_parsers'

// 创建一个 LLM 客户端实例
// 通过环境变量控制 baseURL 和模型,这样换 provider 不用改代码
function createModel() {
  return new ChatOpenAI({
    apiKey: process.env.LLM_API_KEY,
    model: process.env.LLM_MODEL || 'gpt-4o-mini',
    temperature: 0.3,
    configuration: {
      baseURL: process.env.LLM_BASE_URL,  // 比如 https://api.deepseek.com/v1
    },
  })
}

/**
 * 用 LLM 处理一段 prompt,要求输出 JSON
 * @param {string} systemPrompt - 系统提示词
 * @param {string} userPrompt - 用户提示词(含变量已填充)
 * @returns {Promise<object>} 解析后的 JSON 对象
 */
export async function callLLMForJson(systemPrompt, userPrompt) {
  const model = createModel()
  const parser = new JsonOutputParser()
  
  const prompt = ChatPromptTemplate.fromMessages([
    ['system', systemPrompt],
    ['user', '{input}'],
  ])
  
  // LangChain 的链式调用: prompt -> model -> parser
  const chain = prompt.pipe(model).pipe(parser)
  
  return await chain.invoke({ input: userPrompt })
}
为什么这么设计：

createModel() 函数封装了"怎么创建一个 LLM 客户端"，所有配置走环境变量。今天用 DeepSeek，明天换通义千问，改 .env 就行，代码一行不动。
callLLMForJson() 是对外的统一接口——给系统提示和用户提示，返回 JSON 对象。未来你要加重试、加日志、加 token 统计、加多模型 fallback，全在这个函数里改。
用了 LangChain 的 ChatPromptTemplate + pipe() + JsonOutputParser——这就是你学过的那套链式调用，正好用得上。等以后链路复杂（比如先分类再摘要），这套 pipe 写法就能自然扩展。

配置加载
src/config.js
jsimport fs from 'fs/promises'
import path from 'path'
import { parse } from 'yaml'
import { fileURLToPath } from 'url'

const __dirname = path.dirname(fileURLToPath(import.meta.url))

export async function loadTopic(topicId) {
  const filePath = path.join(__dirname, '..', 'config', 'topics', `${topicId}.yaml`)
  const content = await fs.readFile(filePath, 'utf-8')
  const config = parse(content)
  
  // 简单校验,出错时给清楚的提示
  if (!config.id) throw new Error(`Config ${topicId}: missing 'id'`)
  if (!config.title) throw new Error(`Config ${topicId}: missing 'title'`)
  if (!config.sources?.length) throw new Error(`Config ${topicId}: no sources`)
  if (!config.output?.dir) throw new Error(`Config ${topicId}: missing 'output.dir'`)
  
  return config
}
简单校验是为了让错误提早暴露——配置写错了就直接报错，比跑到一半挂掉好排查得多。

抓取与过滤
src/fetch.js
jsimport Parser from 'rss-parser'

const parser = new Parser({ timeout: 15000 })

// 拉一个 RSS 源,转成统一格式
// 单源失败不影响其他源
async function fetchOneSource(source) {
  try {
    const feed = await parser.parseURL(source.url)
    return feed.items.map(item => ({
      title: item.title || '',
      url: item.link || '',
      source: source.name,
      publishedAt: item.isoDate || item.pubDate || new Date().toISOString(),
      summary: item.contentSnippet || item.content || '',
    }))
  } catch (err) {
    console.error(`[fetch] ${source.name} failed: ${err.message}`)
    return []
  }
}

export async function fetchAndFilter(config) {
  // 1. 并发拉取所有源
  const allItems = (await Promise.all(
    config.sources.map(fetchOneSource)
  )).flat()
  
  // 2. 时间窗口过滤
  const cutoff = Date.now() - config.filter.lookbackHours * 3600 * 1000
  const recent = allItems.filter(item => 
    new Date(item.publishedAt).getTime() > cutoff
  )
  
  // 3. 关键词过滤(标题或摘要里命中任意关键词)
  const keywords = config.filter.keywords.map(k => k.toLowerCase())
  const matched = recent.filter(item => {
    const text = `${item.title} ${item.summary}`.toLowerCase()
    return keywords.some(k => text.includes(k))
  })
  
  // 4. URL 去重
  const seen = new Set()
  const deduped = matched.filter(item => {
    if (!item.url || seen.has(item.url)) return false
    seen.add(item.url)
    return true
  })
  
  // 5. 时间倒序,取前 N 条
  deduped.sort((a, b) => new Date(b.publishedAt) - new Date(a.publishedAt))
  return deduped.slice(0, config.filter.maxItems)
}
几个关键设计：

单源失败返回空数组，不会让整体崩溃。这是错误处理的体现。
数据 shape 统一为 {title, url, source, publishedAt, summary}，未来加新源（API、爬虫）只要也产出这个 shape，下游全部不用改。
处理步骤 1-5 是线性的、可读的，没有过度抽象成五个独立文件。等真的某一步要单独测试或替换时再拆。


LLM 摘要
src/summarize.js
jsimport { callLLMForJson } from './llm.js'

const SYSTEM_PROMPT = `你是一名资深的国际新闻编辑。
你的任务是基于提供的新闻条目,生成一份结构化的简报。
严格要求:
- 只基于提供的素材,绝不编造任何事实
- 中文输出
- 严格按照要求的 JSON schema 返回,不要包含任何额外解释`

function buildUserPrompt(items, config) {
  const itemsText = items.map((item, i) =>
    `[${i + 1}] 来源: ${item.source}
时间: ${item.publishedAt}
标题: ${item.title}
摘要: ${item.summary}
URL: ${item.url}`
  ).join('\n\n')
  
  return `请基于下面 ${items.length} 条关于"${config.title}"的新闻,生成简报。

返回 JSON 格式:
{
  "summary": "一句话概览,40 字以内",
  "keyDevelopments": ["关键变化 1", "关键变化 2"],
  "timeline": [{"time": "MM-DD HH:mm", "event": "事件描述"}],
  "risks": ["风险与观察 1"],
  "unknowns": ["信息缺口 1"]
}

要求:
- keyDevelopments 不超过 5 条,按重要性排序
- timeline 按时间正序
- 如果某个字段没有内容,返回空数组

新闻素材:

${itemsText}`
}

export async function summarize(items, config) {
  const userPrompt = buildUserPrompt(items, config)
  return await callLLMForJson(SYSTEM_PROMPT, userPrompt)
}
为什么这样写：

system prompt 和 user prompt 分离——system 是稳定的角色设定，user 是每次变化的具体任务。
这个文件不直接接触 LangChain 或任何 SDK，只通过 callLLMForJson 调用。如果以后改成两阶段处理（先逐条摘要，再综合），改这里加新函数就行，llm.js 完全不动。
prompt 模板写得明确——告诉 LLM "字段没内容就返回空数组"，避免它给你瞎编内容填空。


输出
src/output.js
jsimport fs from 'fs/promises'
import path from 'path'

function todayStr() {
  const d = new Date()
  return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`
}

function buildMarkdown(report, items, config) {
  const date = todayStr()
  const generatedAt = new Date().toISOString()
  
  const keyDev = (report.keyDevelopments || []).map((d, i) => `${i + 1}. ${d}`).join('\n') || '（无）'
  const timeline = (report.timeline || []).map(t => `- ${t.time} ${t.event}`).join('\n') || '（无）'
  const risks = (report.risks || []).map(r => `- ${r}`).join('\n') || '（无）'
  const unknowns = (report.unknowns || []).map(u => `- ${u}`).join('\n') || '（无）'
  const sources = items.map(item => `- [${item.source}: ${item.title}](${item.url})`).join('\n')
  
  return `---
topic: ${config.id}
title: ${config.title}
date: ${date}
generatedAt: ${generatedAt}
sourceCount: ${items.length}
---

# ${config.title}

## 一句话概览

${report.summary || '（无）'}

## 关键变化

${keyDev}

## 时间线

${timeline}

## 风险与观察

${risks}

## 信息缺口

${unknowns}

## 来源

${sources}
`
}

export async function writeOutput(report, items, config) {
  await fs.mkdir(config.output.dir, { recursive: true })
  
  const date = todayStr()
  const baseName = `${date}-${config.title}`
  const mdPath = path.join(config.output.dir, `${baseName}.md`)
  const jsonPath = path.join(config.output.dir, `${baseName}.json`)
  
  const markdown = buildMarkdown(report, items, config)
  const jsonData = {
    topic: config.id,
    title: config.title,
    date,
    generatedAt: new Date().toISOString(),
    ...report,
    sources: items,
  }
  
  await fs.writeFile(mdPath, markdown, 'utf-8')
  await fs.writeFile(jsonPath, JSON.stringify(jsonData, null, 2), 'utf-8')
  
  return { mdPath, jsonPath }
}
关键点：

用日期作文件名 → 同一天跑多次会覆盖，天然幂等。OpenClaw 那边读 ${今天日期}-xxx.md 即可。
同时写 json 和 markdown，json 是数据权威表示，markdown 是衍生呈现。
字段缺失时显示"（无）"而不是报错——LLM 偶尔会偷懒不返回某个字段，要兜住。


CLI 入口
src/index.js
jsimport 'dotenv/config'
import { loadTopic } from './config.js'
import { fetchAndFilter } from './fetch.js'
import { summarize } from './summarize.js'
import { writeOutput } from './output.js'

async function main() {
  const topicId = process.argv[2]
  if (!topicId) {
    console.error('Usage: npm run brief <topic-id>')
    console.error('Example: npm run brief us-iran')
    process.exit(1)
  }
  
  console.log(`[1/4] Loading config for "${topicId}"...`)
  const config = await loadTopic(topicId)
  
  console.log(`[2/4] Fetching news for "${config.title}"...`)
  const items = await fetchAndFilter(config)
  console.log(`      Got ${items.length} items after filter+dedupe`)
  
  if (items.length === 0) {
    console.error('No items to summarize. Aborting.')
    process.exit(1)
  }
  
  console.log('[3/4] Summarizing with LLM...')
  const report = await summarize(items, config)
  
  console.log('[4/4] Writing output...')
  const { mdPath, jsonPath } = await writeOutput(report, items, config)
  console.log(`      ✓ ${mdPath}`)
  console.log(`      ✓ ${jsonPath}`)
  console.log('Done.')
}

main().catch(err => {
  console.error('FAILED:', err.message)
  console.error(err.stack)
  process.exit(1)
})
为什么这样写：

四个步骤分别打印，跑的时候你能看到进度，挂在哪一步一目了然。
items.length === 0 时直接退出，不写文件——这样昨天的简报不会被今天的失败污染。
process.exit(1) 让 OpenClaw 的 cron 能感知到失败。

未来扩展时改哪里
我把每个常见扩展场景对应到具体改哪个文件，你和 Codex 都能一眼看明白：
加新主题 → 只新建一个 config/topics/xxx.yaml，代码不动。
加新 RSS 源 → 改对应主题的 yaml，加一项 sources，代码不动。
加非 RSS 类型的源（API/爬虫） → 在 fetch.js 里加一个函数（比如 fetchFromApi），在 yaml 里给 source 加 type 字段，dispatcher 在 fetch.js 里分发。这时候才有理由把 fetch.js 拆成 sources/rss.js、sources/api.js。
换 LLM provider → 改 .env 里的 LLM_BASE_URL 和 LLM_MODEL，代码不动。
改 prompt → 改 summarize.js 里的 SYSTEM_PROMPT 和 buildUserPrompt。
加链式处理（先分类再摘要） → 在 llm.js 里加新的导出函数（比如 callLLMChain），用 LangChain 的 RunnableSequence 编排。summarize.js 调用新函数。
加 OpenClaw 推送 → 不改任何东西。OpenClaw 那边写一段读取 ${今天日期}-xxx.md 然后推飞书的逻辑就行。这正是解耦的好处——两个系统通过文件系统通信，互不知道对方存在。
加运行历史归档 → 在 index.js 里加一段，把每次运行的中间结果（fetch 后的 items、LLM 返回的 report）写到 data/runs/${topicId}/${timestamp}/。等真的要排查 bug 时再加。
